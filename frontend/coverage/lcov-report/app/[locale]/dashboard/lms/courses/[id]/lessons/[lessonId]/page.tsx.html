                <div className="text-right">

              </Link>
            ) : (
              <Link
                href={`/dashboard/lms/courses/${courseId}`}
                className="flex items-center gap-2 px-4 py-2 bg-green-600 text-white rounded-lg hover:bg-green-700"
              >
                <CheckCircle className="h-4 w-4" />
                Finalizeaza cursul
              </Link>
            )}
          </div>
        </div>
      </div>
    </div>
  );
